2.1 Conceptual ER Diagram and Relational Schema Translation

I designed my project in the `soccer_proj` schema with 7 relational tables and 1 view:
1) Country
2) League
3) Team
4) Player
5) Team_Attributes
6) Player_Attributes
7) Match_tbl
View: league_match_summary_view

Relational schema (PK, UNIQUE, FK):

Country(
  id PK,
  name
)

League(
  id PK,
  country_id FK -> Country(id),
  name
)

Team(
  id PK,
  team_api_id UNIQUE,
  team_fifa_api_id UNIQUE,
  team_long_name,
  team_short_name
)

Player(
  id PK,
  player_api_id UNIQUE,
  player_fifa_api_id UNIQUE,
  player_name,
  birthday,
  height,
  weight
)

Team_Attributes(
  id PK,
  team_api_id FK -> Team(team_api_id),
  team_fifa_api_id FK -> Team(team_fifa_api_id),
  date,
  buildUpPlaySpeed,
  buildUpPlayPassing,
  chanceCreationPassing,
  defencePressure
  -- plus additional team attribute columns
)

Player_Attributes(
  id PK,
  player_api_id FK -> Player(player_api_id),
  player_fifa_api_id FK -> Player(player_fifa_api_id),
  date,
  overall_rating,
  potential
  -- plus additional player attribute columns
)

Match_tbl(
  id PK,
  country_id FK -> Country(id),
  league_id FK -> League(id),
  home_team_api_id FK -> Team(team_api_id),
  away_team_api_id FK -> Team(team_api_id),
  home_player_1 ... home_player_11 FK -> Player(player_api_id),
  away_player_1 ... away_player_11 FK -> Player(player_api_id),
  match_api_id,
  season,
  stage,
  date,
  home_team_goal,
  away_team_goal
  -- plus betting/event columns
)

View created:

create or replace view soccer_proj.league_match_summary_view as
select
  l.id as league_id,
  l.name as league_name,
  count(*) as total_matches,
  avg(m.home_team_goal)::numeric(10,2) as avg_home_goals,
  avg(m.away_team_goal)::numeric(10,2) as avg_away_goals,
  avg((m.home_team_goal + m.away_team_goal)::numeric(10,2)) as avg_total_goals
from soccer_proj."Match_tbl" m
join soccer_proj."League" l
  on l.id = m.league_id
group by l.id, l.name;
