Self Joins
A self join is a regular join, but the table is joined with itself.
create schema if not exists self_join;
create table if not exists self_join.employees (
employee_id int
, last_name text
, first_name text
, title text
, department_id int
, manager_id int -- the employee_id for the manager
constraint manager_fk
references self_join.employees(employee_id)
deferrable initially deferred
, constraint employees_pk primary key (employee_id)
);

, worker.title
FROM self_join.employees as worker

create table if not exists self_join.employees (
    employee_id int
    , last_name text
    , first_name text
    , title text
    , department_id int
    , manager_id int -- the employee_id for the manager
   constraint manager_fk
        references self_join.employees(employee_id)
        deferrable initially deferred
    , constraint employees_pk primary key (employee_id)
    , constraint dept_fk foreign key (department_id)
      references self_join.departments(department_id)
);
